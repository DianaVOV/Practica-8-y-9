#include<stdio.h>

int main(){
	int matriz[1000][1000],renglones,columnas,i,j;
	
	printf("Digite el numero de renglones: ");
	scanf("%i",&renglones);
	printf("Digite el numero de columnas: ");
	scanf("%i",&columnas);
	
	printf("\n\n");
	
	for(i=0;i<renglones;i++){
		for(j=0;j<columnas;j++){
			printf("Digite un numero matriz[%i][%i]: ",i+1,j+1);
			scanf("%i",&matriz[i][j]);
		}
	}
	
	printf("\n\n");
	
	for(i=0;i<renglones;i++){
		for(j=0;j<columnas;j++){
			printf("%i ",matriz[i][j]);
		}
		printf("\n");
	}
}
